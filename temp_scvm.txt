using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using DiskChecker.Core.Interfaces;
using DiskChecker.Core.Models;
using DiskChecker.UI.Avalonia.Services.Interfaces;
using System;
using System.Collections.ObjectModel;
using System.Threading.Tasks;

namespace DiskChecker.UI.Avalonia.ViewModels
{
    public partial class SmartCheckViewModel : ViewModelBase
    {
        private readonly ISmartService _smartService;
        private readonly IDialogService _dialogService;
        private ObservableCollection<DiskInfo> _disks = new();
        private DiskInfo? _selectedDisk;
        private bool _isChecking;
        private string _statusMessage = string.Empty;

        public SmartCheckViewModel(ISmartService smartService, IDialogService dialogService)
        {
            _smartService = smartService ?? throw new ArgumentNullException(nameof(smartService));
            _dialogService = dialogService ?? throw new ArgumentNullException(nameof(dialogService));
            
            LoadDisksCommand = new AsyncRelayCommand(LoadDisksAsync);
            CheckSmartCommand = new AsyncRelayCommand(CheckSmartAsync, () => SelectedDisk != null && !IsChecking);
            LoadDisksCommand.Execute(null); // Načteme disky hned při startu
        }

        public ObservableCollection<DiskInfo> Disks
        {
            get => _disks;
            set => SetProperty(ref _disks, value);
        }

        public DiskInfo? SelectedDisk
        {
            get => _selectedDisk;
            set
            {
                if (SetProperty(ref _selectedDisk, value))
                {
                    CheckSmartCommand.NotifyCanExecuteChanged();
                }
            }
        }

        public bool IsChecking
        {
            get => _isChecking;
            set
            {
                if (SetProperty(ref _isChecking, value))
                {
                    CheckSmartCommand.NotifyCanExecuteChanged();
                }
            }
        }

        public string StatusMessage
        {
            get => _statusMessage;
            set => SetProperty(ref _statusMessage, value);
        }

        public IAsyncRelayCommand LoadDisksCommand { get; }
        public IAsyncRelayCommand CheckSmartCommand { get; }

        private async Task LoadDisksAsync()
        {
            try
            {
                StatusMessage = "Načítám seznam disků...";
                var disks = await _smartService.GetDisksWithSmartAsync();
                Disks = new ObservableCollection<DiskInfo>(disks);
                StatusMessage = $"Načteno {disks.Count} disků";
            }
            catch (Exception ex)
            {
                StatusMessage = $"Chyba při načítání disků: {ex.Message}";
                await _dialogService.ShowErrorAsync("Chyba", $"Nepodařilo se načíst seznam disků: {ex.Message}");
            }
        }

        private async Task CheckSmartAsync()
        {
            if (SelectedDisk == null) return;

            try
            {
                IsChecking = true;
                StatusMessage = $"Kontroluji SMART data pro {SelectedDisk.Model} ({SelectedDisk.SerialNumber})...";
                
                var smartData = await _smartService.GetSmartDataAsync(SelectedDisk);
                
                // Zde bychom normálně zobrazili detailní výsledky SMART kontroly
                StatusMessage = "SMART kontrola dokončena";
                
                // Pro zjednodušení zatím jen ukážeme dialog s výsledkem
                var healthStatus = smartData?.Temperature > 0 ? "Zdravý" : "Neznámý";
                await _dialogService.ShowMessageAsync("Výsledek SMART kontroly", 
                    $"Disk: {SelectedDisk.Model}\n" +
                    $"Serial: {SelectedDisk.SerialNumber}\n" +
                    $"Teplota: {smartData?.Temperature ?? 0}°C\n" +
                    $"Stav: {healthStatus}");
            }
            catch (Exception ex)
            {
                StatusMessage = $"Chyba při kontrole SMART: {ex.Message}";
                await _dialogService.ShowErrorAsync("Chyba", $"Nepodařilo se provést SMART kontrolu: {ex.Message}");
            }
            finally
            {
                IsChecking = false;
            }
        }
    }
}
